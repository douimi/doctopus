# Doctopus Plan 1.B — Patients & Day View Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build patient records and the front-desk day view so a Moroccan generalist's cabinet can manage demographics, schedule appointments, handle walk-ins, and run the waiting-room workflow end-to-end. No consultation editor or prescriptions yet.

**Architecture:** All new tables (`patients`, `patient_allergies`, `patient_chronic_conditions`, `appointments`) are tenant-scoped and protected by Row Level Security. Read paths are React Server Components; write paths are Server Actions. Every DB access goes through `withTenantTx` from Plan 1.A. The `/today` page is the cabinet's home: a server-rendered schedule + waiting-room view with Server-Action-driven status transitions, and a small set of focused sub-routes for walk-in and booking flows (no client-side dialogs needed).

**Tech Stack:** Next.js 16 App Router, React 19 (Server Components + Server Actions, `useActionState`), Drizzle ORM, Supabase Postgres + Auth, Tailwind v4 + shadcn/ui (button, input, label, card; adding `select` and `textarea` in this plan), zod for input validation, Vitest + Playwright.

**Spec reference:** [`docs/superpowers/specs/2026-04-30-doctopus-mvp-design.md`](../specs/2026-04-30-doctopus-mvp-design.md), Sections 4 (patients, appointments), 5.B (front-desk), 8 (audit log fields prepared but writes deferred to Plan 1.E).

**Prerequisites:** Plan 1.A complete on `main` (tag `plan-1a-foundation`). Supabase local stack running.

## What Plan 1.B does NOT do (carry-forward to later plans)

- No consultation editor (`/consultations/[id]`) — Plan 1.C.
- No prescription editor or PDF — Plan 1.D.
- No `audit_log` table or audit writes — Plan 1.E. (Patient/appointment Server Actions get one-line "audit hook" stubs we wire up in 1.E.)
- No "Commencer la consultation" button on `/today` — that needs the consultation editor; we render a placeholder badge for `in_consultation` only.

## Conventions every task follows

- pnpm-only. Conventional Commits prefixes (`feat`, `test`, `refactor`, `chore`, `docs`).
- TDD where there's logic to test (state machine, age helper, RLS isolation, queries with non-trivial filters).
- Pure scaffolding/UI tasks may not have unit tests; they're covered by E2E.
- Every task ends on a clean working tree with one commit (unless the task says otherwise).
- Every Server Action: validates input with zod, then loads the session and calls `withTenantTx` (or, where the action is doctor-only, calls `requireDoctor` first).

**Carry-forward caveats from 1.A you must remember:**
- Use **`useActionState` from `react`** (not `useFormState` from `react-dom`).
- Wrap any client component using **`useSearchParams`** in a `<Suspense>` boundary or the production build will fail.
- After every `pnpm install`, restore the Supabase CLI binary: `cp node_modules/.pnpm/supabase@*/node_modules/supabase/bin/supabase.exe node_modules/.bin/supabase.exe` (Windows; harmless to skip on Linux).
- `db/client.ts` is shared between Server Components and CLI/test code, so it does **not** import `'server-only'`. New helpers in `lib/patients/queries.ts`, `lib/patients/mutations.ts`, etc. **do** import `'server-only'`.

```ts
import { boolean, date, pgTable, text, timestamp, uuid } from 'drizzle-orm/pg-core';
import { tenants } from './tenants';

export const patients = pgTable('patients', {
  id: uuid('id').primaryKey().defaultRandom(),
  tenantId: uuid('tenant_id').notNull().references(() => tenants.id, { onDelete: 'restrict' }),
  firstName: text('first_name').notNull(),
  lastName: text('last_name').notNull(),
  gender: text('gender', { enum: ['m', 'f'] }).notNull(),
  dateOfBirth: date('date_of_birth').notNull(),
  phone: text('phone'),
  cin: text('cin'),
  coverageType: text('coverage_type', {
    enum: ['cnss', 'cnops', 'amo', 'ramed', 'mutuelle', 'none', 'other'],
  }),
  coverageId: text('coverage_id'),
  address: text('address'),
  notes: text('notes'),
  isArchived: boolean('is_archived').notNull().default(false),
  createdAt: timestamp('created_at', { withTimezone: true }).notNull().defaultNow(),
  updatedAt: timestamp('updated_at', { withTimezone: true }).notNull().defaultNow(),
});

export const patientAllergies = pgTable('patient_allergies', {
  id: uuid('id').primaryKey().defaultRandom(),
  tenantId: uuid('tenant_id').notNull().references(() => tenants.id, { onDelete: 'restrict' }),
  patientId: uuid('patient_id').notNull().references(() => patients.id, { onDelete: 'cascade' }),
  label: text('label').notNull(),
  createdAt: timestamp('created_at', { withTimezone: true }).notNull().defaultNow(),
});

export const patientChronicConditions = pgTable('patient_chronic_conditions', {
  id: uuid('id').primaryKey().defaultRandom(),
  tenantId: uuid('tenant_id').notNull().references(() => tenants.id, { onDelete: 'restrict' }),
  patientId: uuid('patient_id').notNull().references(() => patients.id, { onDelete: 'cascade' }),
  label: text('label').notNull(),
  createdAt: timestamp('created_at', { withTimezone: true }).notNull().defaultNow(),
});

export type Patient = typeof patients.$inferSelect;
export type NewPatient = typeof patients.$inferInsert;
export type PatientAllergy = typeof patientAllergies.$inferSelect;
export type PatientChronicCondition = typeof patientChronicConditions.$inferSelect;
```

```ts
import { withTenantTx } from '@/db/with-tenant';

    const a = await seedTenant('rls-p-a');
    const b = await seedTenant('rls-p-b');
    await seedPatient(a.tenantId, { firstName: 'PA' });

    const b = await seedTenant('rls-p-d');

    const b = await seedTenant('rls-p-f');
```

```ts
export function ageFromDob(dob: string, now: Date = new Date()): number {
  const [y, m, d] = dob.split('-').map(Number);
  let age = now.getUTCFullYear() - y;
  const month = now.getUTCMonth() + 1;
  const day = now.getUTCDate();
  if (month < m || (month === m && day < d)) age -= 1;
  return age;
}
```

```ts
import 'server-only';
import { and, desc, eq, ilike, or, sql } from 'drizzle-orm';
import { withTenantTx } from '@/db/with-tenant';
import { patients, patientAllergies, patientChronicConditions } from '@/db/schema';

export type PatientListRow = {
  id: string;
  firstName: string;
  lastName: string;
  phone: string | null;
  cin: string | null;
  dateOfBirth: string;
  isArchived: boolean;
};

const ESCAPE_PATTERN = /[\\%_]/g;
function escapeIlike(input: string): string {
  return input.replace(ESCAPE_PATTERN, (m) => '\\' + m);
}

export async function searchPatients(
  tenantId: string,
  query: string,
  opts: { includeArchived?: boolean; limit?: number } = {},
): Promise<PatientListRow[]> {
  const limit = Math.min(opts.limit ?? 50, 200);
  const trimmed = query.trim();
  return withTenantTx(tenantId, async (tx) => {
    const where = trimmed
      ? and(
          opts.includeArchived ? undefined : eq(patients.isArchived, false),
          or(
            ilike(patients.firstName, `%${escapeIlike(trimmed)}%`),
            ilike(patients.lastName, `%${escapeIlike(trimmed)}%`),
            ilike(patients.phone, `%${escapeIlike(trimmed)}%`),
            ilike(patients.cin, `%${escapeIlike(trimmed)}%`),
          ),
        )
      : opts.includeArchived
      ? undefined
      : eq(patients.isArchived, false);

    const rows = await tx
      .select({
        id: patients.id,
        firstName: patients.firstName,
        lastName: patients.lastName,
        phone: patients.phone,
        cin: patients.cin,
        dateOfBirth: patients.dateOfBirth,
        isArchived: patients.isArchived,
      })
      .from(patients)
      .where(where)
      .orderBy(desc(patients.createdAt))
      .limit(limit);
    return rows;
  });
}

export async function getPatientById(tenantId: string, id: string) {
  return withTenantTx(tenantId, async (tx) => {
    const [row] = await tx.select().from(patients).where(eq(patients.id, id));
    return row ?? null;
  });
}

export async function getPatientDetail(tenantId: string, id: string) {
  return withTenantTx(tenantId, async (tx) => {
    const [patient] = await tx.select().from(patients).where(eq(patients.id, id));
    if (!patient) return null;
    const allergies = await tx
      .select()
      .from(patientAllergies)
      .where(eq(patientAllergies.patientId, id))
      .orderBy(patientAllergies.label);
    const conditions = await tx
      .select()
      .from(patientChronicConditions)
      .where(eq(patientChronicConditions.patientId, id))
      .orderBy(patientChronicConditions.label);
    return { patient, allergies, conditions };
  });
}

// Used by appointment queries (avoids N+1 by batching)
export async function getPatientsForAppointmentList(tenantId: string, patientIds: string[]) {
  if (patientIds.length === 0) return new Map<string, typeof patients.$inferSelect>();
  return withTenantTx(tenantId, async (tx) => {
    const rows = await tx
      .select()
      .from(patients)
      .where(sql`${patients.id} = ANY(${sql.raw(`ARRAY['${patientIds.join("','")}']::uuid[]`)})`);
    return new Map(rows.map((r) => [r.id, r]));
  });
}
```

```ts
export type AppointmentStatus =
  | 'scheduled'
  | 'waiting'
  | 'in_consultation'
  | 'done'
  | 'cancelled'
  | 'no_show';

export type Action = 'arrive' | 'start' | 'finalize' | 'cancel' | 'noShow';

const TRANSITIONS: Record<AppointmentStatus, Partial<Record<Action, AppointmentStatus>>> = {
  scheduled: { arrive: 'waiting', cancel: 'cancelled', noShow: 'no_show' },
  waiting: { start: 'in_consultation', cancel: 'cancelled' },
  in_consultation: { finalize: 'done' },
  done: {},
  cancelled: {},
  no_show: {},
};

export function canTransition(from: AppointmentStatus, action: Action): boolean {
  return TRANSITIONS[from][action] !== undefined;
}

export type AppointmentPatch = {
  status: AppointmentStatus;
  arrivedAt?: Date;
  startedAt?: Date;
  endedAt?: Date;
};

export function applyTransition(
  from: AppointmentStatus,
  action: Action,
  now: Date,
): AppointmentPatch {
  const next = TRANSITIONS[from][action];
  if (!next) {
    throw new Error(`Transition not allowed: ${from} --${action}-->`);
  }
  switch (action) {
    case 'arrive':
      return { status: next, arrivedAt: now };
    case 'start':
      return { status: next, startedAt: now };
    case 'finalize':
      return { status: next, endedAt: now };
    case 'cancel':
    case 'noShow':
      return { status: next };
  }
}
```

```ts
    const patch = applyTransition(current.status, action, new Date());
    const [updated] = await tx
      .update(appointments)
      .set({
        status: patch.status,
        arrivedAt: patch.arrivedAt ?? current.arrivedAt,
        startedAt: patch.startedAt ?? current.startedAt,
        endedAt: patch.endedAt ?? current.endedAt,
        updatedAt: new Date(),
      })
      .where(eq(appointments.id, id))
      .returning();
    return updated;
  });
}

  transitionAppointment(tenantId, id, 'arrive');
export const cancelAppointment = (tenantId: string, id: string) =>
  transitionAppointment(tenantId, id, 'cancel');
export const markNoShow = (tenantId: string, id: string) =>
```

```ts
    patientId: formData.get('patientId'),
    scheduledAt,
    reason: formData.get('reason'),
  });
  if (!parsed.success) return { error: 'Données invalides.' };
```
